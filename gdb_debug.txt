# GDB调试命令文件
# 使用方法: gdb -x gdb_debug.txt kernel.bin

# 连接到QEMU
target remote :1234

# 设置架构
set architecture i386

# 设置断点
break start_task_user
break user_task_main
break task_to_user_mode

# 显示断点信息
info breakpoints

# 继续执行
continue

# 当停在断点时,使用以下命令:
# si           - 单步执行汇编指令
# disassemble  - 查看反汇编代码
# info registers - 显示寄存器
# print th_u    - 打印th_u指针
# print *th_u->tf - 打印trapframe内容
# print/x $cr3 - 打印CR3寄存器
# x/20x $esp    - 显示栈内容
